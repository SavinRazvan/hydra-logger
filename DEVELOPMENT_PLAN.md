from hydra_logger import HydraLogger

# Built-in framework magic

@HydraLogger.register_magic("my_app")
def my_app_config():
    return LoggingConfig(layers={"APP": LogLayer(...)})

# Common magic configs
logger = HydraLogger.for_web_app()      # Web application
